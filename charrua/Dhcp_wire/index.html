  1. srcmac : Macaddr.t;
  2. dstmac : Macaddr.t;
  3. srcip : Ipaddr.V4.t;
  4. dstip : Ipaddr.V4.t;
  5. srcport : int;
  6. dstport : int;
  7. op : op;
  8. htype : htype;
  9. hlen : int;
  10. hops : int;
  11. xid : int32;
  12. secs : int;
  13. flags : flags;
  14. ciaddr : Ipaddr.V4.t;
  15. yiaddr : Ipaddr.V4.t;
  16. siaddr : Ipaddr.V4.t;
  17. giaddr : Ipaddr.V4.t;
  18. chaddr : Macaddr.t;
  19. sname : string;
  20. file : string;
  21. options : dhcp_option list;
}
